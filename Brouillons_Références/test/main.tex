\documentclass{beamer}
\usepackage{tikz}

% Définir une couleur personnalisée
\definecolor{customBleuSombre}{rgb}{0, 0.285, 0.563}

% Définir la commande pour la barre de progression
\newcommand{\progressbar}{
  \tikz[remember picture,overlay]{
    \fill[customBleuSombre!40] (current page.south west) rectangle ++(\paperwidth*\insertframenumber/\inserttotalframenumber,4pt);
  }
}

\begin{document}

% Ajouter la barre de progression à chaque diapositive
\addtobeamertemplate{footline}{}{
  \progressbar
}
